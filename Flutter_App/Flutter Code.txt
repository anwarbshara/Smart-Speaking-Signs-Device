class SmartGloveApp extends StatefulWidget {
  List<String> gestureHistory = [];

  @override
  State<SmartGloveApp> createState() => _SmartGloveAppState();
}

class _SmartGloveAppState extends State<SmartGloveApp> {
  Future<bool> requestBluetoothPermissions() async {
    final status = await [
      Permission.bluetooth,
      Permission.bluetoothScan,
      Permission.bluetoothConnect,
      Permission.bluetoothAdvertise,
      Permission.locationWhenInUse,
    ].request();

    return status.values.every((perm) => perm.isGranted);
  }

  late BuildContext globalContext;
  bool isDarkMode = false;
  bool isSoundOn = true;
  Locale _locale = Locale('en');
  String btStatus = 'Checking...';
  String currentSentence = '';
  List<String> gestureHistory = [];

  final FlutterTts tts = FlutterTts();
  final dbRef = FirebaseDatabase.instance.ref("gestures");

  @override
  void initState() {
    super.initState();
    listenToBluetoothStatus();
    connectToESP32();
  }

  void clearGestureHistory() {
    setState(() {
      gestureHistory.clear();
      currentSentence = '';
    });
  }

  void deleteGestureAt(int index) {
    setState(() {
      gestureHistory.removeAt(index);
      currentSentence = gestureHistory.join(' ');
    });
  }


  void clearCurrentSentence() {
    setState(() {
      currentSentence = '';
    });
  }

  Future<void> requestPermissions() async {
    await [
      Permission.bluetoothScan,
      Permission.bluetoothConnect,
      Permission.locationWhenInUse,
    ].request();
  }

  void listenToBluetoothStatus() {
    FlutterBluePlus.adapterState.listen((state) {
      String statusKey;
      switch (state) {
        case BluetoothAdapterState.on:
          statusKey = 'connected';
          break;
        case BluetoothAdapterState.off:
          statusKey = 'disconnected';
          break;
        case BluetoothAdapterState.unavailable:
          statusKey = 'unavailable';
          break;
        case BluetoothAdapterState.unauthorized:
          statusKey = 'unauthorized';
          break;
        case BluetoothAdapterState.turningOn:
          statusKey = 'turning_on';
          break;
        case BluetoothAdapterState.turningOff:
          statusKey = 'turning_off';
          break;
        default:
          statusKey = 'unknown';
      }

      if (mounted) {
        setState(() {
          btStatus = statusKey; // المفتاح وليس الترجمة نفسها
        });
      }
    });
  }


  void connectToESP32() async {
    final granted = await requestBluetoothPermissions();
    if (!granted) {
      print("🚫 الصلاحيات غير مفعّلة! لن يتم بدء الفحص.");
      return;
    }
    final serviceUuid = Guid("4fafc201-1fb5-459e-8fcc-c5c9c331914b");
    final characteristicUuid = Guid("beb5483e-36e1-4688-b7f5-ea07361b26a8");
    FlutterBluePlus.startScan(timeout: Duration(seconds: 5));
    FlutterBluePlus.scanResults.listen((results) async {
      for (var result in results) {
        print(
            "📡 Found device: ${result.device.name} | id: ${result.device.id}");
        if (result.device.id.toString().trim() == "A0:B7:65:28:35:5A" ||
            result.device.name == "SSSD-ESP32") {
          await FlutterBluePlus.stopScan();
          print("✅ Connecting to SSSD-ESP32...");

          try {
            await result.device.connect(autoConnect: false);
          } catch (e) {
            print("⚠️ Connection error: $e");
            return;
          }

          final services = await result.device.discoverServices();
          print("🔍 Services discovered: ${services.length}");

          for (var service in services) {
            if (service.uuid == serviceUuid) {
              for (var characteristic in service.characteristics) {
                if (characteristic.uuid == characteristicUuid &&
                    characteristic.properties.notify) {
                  print(
                      "📡 Found notify characteristic: ${characteristic.uuid}");
                  await characteristic.setNotifyValue(true);
                  characteristic.onValueReceived.listen((value) {
                    final code = String.fromCharCodes(value).trim();
                    print("🛰️ Received from ESP32: $code");
                    handleGestureCode(code);
                  });
                }
              }
            }
          }
        }
      }
    });
  }

  void handleGestureCode(String code) async {
    final langCode = Localizations.localeOf(globalContext).languageCode;

    if (code == "000000") {
      final noMatchText = langCode == 'ar' ? "❌ لا يوجد تطابق" : "❌ No match found";

      WidgetsBinding.instance.addPostFrameCallback((_) {
        ScaffoldMessenger.of(globalContext).showSnackBar(
          SnackBar(
            content: Text(noMatchText),
            backgroundColor: Colors.redAccent,
            duration: Duration(seconds: 2),
          ),
        );
      });



      return;
    }
    final snapshot = await dbRef.child(code).get();
    if (snapshot.exists) {
      final textKey = langCode == 'ar' ? 'text_ar' : 'text_en';
      final text = snapshot
          .child(textKey)
          .value
          .toString();

      setState(() {
        if (gestureHistory.isEmpty || gestureHistory.last != text) {
          gestureHistory.add(text);
          currentSentence = gestureHistory.join(' ');
        }
      });


      /*if (isSoundOn) {
        await tts.setLanguage(_locale.languageCode);
        await tts.setSpeechRate(0.5);
        await tts.speak(text);
      }
    }
    else {
      setState(() {
        if (!currentSentence.endsWith("($code)")) {
          setState(() {
            currentSentence += (currentSentence.isEmpty ? '' : ' ') + "إشارة غير معروفة ($code)";
          });
        }
      });
    }*/
    }
  }

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      locale: _locale,
      supportedLocales: AppLocalizations.supportedLocales,
      localizationsDelegates: AppLocalizations.localizationsDelegates,
      debugShowCheckedModeBanner: false,
      theme: isDarkMode
          ? ThemeData.dark(useMaterial3: true)
          : ThemeData.light(useMaterial3: true).copyWith(
        scaffoldBackgroundColor: const Color(0xFFF2F7FA),
        colorScheme: ColorScheme.fromSeed(seedColor: Colors.teal),
        textTheme: ThemeData
            .light()
            .textTheme
            .apply(fontFamily: 'Cairo'),
      ),
      home: Builder(
        builder: (context) {
          globalContext = context;
          return HomePage(
            isDarkMode: isDarkMode,
            onToggleTheme: () => setState(() => isDarkMode = !isDarkMode),
            isSoundOn: isSoundOn,
            onToggleSound: () => setState(() => isSoundOn = !isSoundOn),
            currentLocale: _locale,
            onClearCurrentSentence: clearCurrentSentence,
            onLanguageChange: (langCode) {
              setState(() {
                _locale = Locale(langCode);
              });
            },
            btStatusKey: btStatus,

            gestureHistory: gestureHistory,
            onClearGestureHistory: clearGestureHistory,
            currentSentence: currentSentence,
            onDeleteGestureAt: deleteGestureAt,
          );
        },
      ),
    );
  }
}

class GestureHistoryScreen extends StatefulWidget {
  final List<String> history;
  final Function(int) onDeleteGestureAt;

  const GestureHistoryScreen({
    Key? key,
    required this.history,
    required this.onDeleteGestureAt,
  }) : super(key: key);

  @override
  State<GestureHistoryScreen> createState() => _GestureHistoryScreenState();
}

class _GestureHistoryScreenState extends State<GestureHistoryScreen> {
  @override
  Widget build(BuildContext context) {
    final tr = AppLocalizations.of(context)!;

    return Scaffold(
      appBar: AppBar(
        title: Text(tr.history_g),
        backgroundColor: Colors.teal,
      ),
      body: Container(
        decoration: BoxDecoration(
          gradient: LinearGradient(
            begin: Alignment.topCenter,
            end: Alignment.bottomCenter,
            colors: [Color(0xFF2CA7A2), Color(0xFFF2F7FA)],
          ),
        ),
        child: widget.history.isEmpty
            ? Center(
          child: Text(
            tr.no_gesture_yet,
            style: TextStyle(fontSize: 18, color: Colors.white),
          ),
        )
            : ListView.builder(
          padding: EdgeInsets.all(16),
          itemCount: widget.history.length,
          itemBuilder: (context, index) {
            final isNoMatch = widget.history[index].contains("❌");
            final gesture = widget.history[index];

            return Dismissible(
              key: Key(gesture + index.toString()),
              direction: DismissDirection.endToStart,
              background: Container(
                padding: EdgeInsets.only(right: 20),
                alignment: Alignment.centerRight,
                color: Colors.redAccent,
                child: Icon(Icons.delete, color: Colors.white),
              ),
              confirmDismiss: (_) async {
                final result = await showDialog<bool>(
                  context: context,
                  builder: (_) => AlertDialog(
                    title: Text(tr.clear_gestures),
                    content: Text("${tr.history_g}:\n\"$gesture\"؟"),
                    actions: [
                      TextButton(
                        onPressed: () => Navigator.of(context).pop(false),
                        child: Text(AppLocalizations.of(context)!.cancel ?? 'Cancel'),
                      ),
                      TextButton(
                        onPressed: () => Navigator.of(context).pop(true),
                        child: Text(tr.clear ?? 'Clear'),
                      ),
                    ],
                  ),
                );

                return result == true;
              },
                onDismissed: (_) {
                  widget.onDeleteGestureAt(index); // بدون setState هنا
                  ScaffoldMessenger.of(context).showSnackBar(
                    SnackBar(
                      content: Text('✅ "$gesture" ${tr.clear_gestures}'),
        
                      backgroundColor: Colors.teal,
                    ),
                  );
                },
              child: Card(
                shape: RoundedRectangleBorder(
                  borderRadius: BorderRadius.circular(16),
                ),
                elevation: 4,
                margin: EdgeInsets.symmetric(vertical: 8),
                child: ListTile(
                  leading: CircleAvatar(
                    backgroundColor: isNoMatch ? Colors.red : Colors.teal,
                    child: Icon(
                      isNoMatch ? Icons.block : Icons.handshake,
                      color: Colors.white,
                    ),
                  ),
                  title: Text(
                    gesture,
                    style: TextStyle(
                      fontSize: 16,
                      fontWeight: FontWeight.w600,
                      color: isNoMatch ? Colors.red : Colors.black,
                    ),
                  ),
                  trailing: isNoMatch
                      ? Icon(Icons.warning_amber_rounded, color: Colors.redAccent)
                      : Icon(Icons.check_circle_outline, color: Colors.teal),
                ),
              ),
            );
          },
        ),
      ),
    );
  }
}
